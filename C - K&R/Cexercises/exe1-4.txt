#include <stdio.h>

#define LOWER 0
#define UPPER 150
#define STEP 20

void print_conversao();

int main()
{
	printf("CELSIUS \t FAHRENHEIT");
	print_conversao();
}

void print_conversao(){
	float cels = LOWER;
	while(cels < UPPER){
		float fahr = (9.0/5.0) * cels + 32;
		printf("\n%3.0f \t\t %6.2f", cels, fahr);
		cels += STEP;
	}
}
