#include <stdio.h>

#define LOWER 0
#define UPPER 300
#define STEP 20

void print_conversao();

int main()
{
	printf("FAHRENHEIT \t CELSIUS");
	print_conversao();
}

void print_conversao(){
	//float fahr = UPPER;
	for(float fahr = UPPER; fahr >= LOWER; fahr -= STEP){
		printf("\n%3.0f \t\t %6.2f", fahr, (5.0/9.0) * (fahr - 32));
	}
}
